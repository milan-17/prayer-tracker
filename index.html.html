PrayerRepositoryImpl.kt

Business Logic Layer 
class PrayerRepositoryImpl @Inject constructor(
    private val prayerDao: PrayerDao,
    private val prayerLogDao: PrayerLogDao
) : PrayerRepository {

    override suspend fun logPrayer(
        prayerId: Long, 
        isPerformed: Boolean, 
        notes: String?, 
        timestamp: Instant
    ) {
        val (start, end) = DateUtils.getDayStartAndEnd(timestamp)
        val existingLog = prayerLogDao.getLogForPrayerAndDay(prayerId, start, end)
        
        if (existingLog != null) {
            prayerLogDao.updatePrayerLog(
                existingLog.copy(isPerformed = isPerformed, notes = notes)
            )
        } else {
            prayerLogDao.insertPrayerLog(
                PrayerLogEntity(prayerId = prayerId, isPerformed = isPerformed, timestamp = timestamp)
            )
        }
    }
}

@AndroidEntryPoint
class MainActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContent {
            val viewModel: MainViewModel = hiltViewModel()
            val isDarkMode by viewModel.isDarkMode.collectAsStateWithLifecycle()

            OraTrackTheme(darkTheme = isDarkMode) {
                OraTrackAppContent()
            }
        }
    }
}

@Composable
fun OraTrackAppContent() {
    val navController = rememberNavController()
    Scaffold(
        bottomBar = { BottomNavigationBar(navController) }
    ) { padding ->
        OraTrackNavHost(navController, Modifier.padding(padding))
    }
}

Core System Features

Advanced Analytics

Integrated MPAndroidChart for visualizing prayer consistency over a 30-day window with high-performance data binding.

Dynamic Scheduling

Leverages WorkManager for precise notification delivery and periodic database backups even when the app is inactive.

Secure Backup

Full SQLite database export and import using Android's Storage Access Framework for user-controlled data ownership.

State Management

Robust DataStore integration for persistent user preferences like Theme Selection and notification intervals.
